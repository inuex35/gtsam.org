160 inline static Rot3 RzRyRx(const Vector& xyz,
161 OptionalJacobian<3, 3> H = boost::none) {
162 assert(xyz.size() == 3);
163 Rot3 out;
164 if (H) {
165 Vector3 Hx, Hy, Hz;
166 out = RzRyRx(xyz(0), xyz(1), xyz(2), Hx, Hy, Hz);
167 (*H) << Hx, Hy, Hz;
168 } else
169 out = RzRyRx(xyz(0), xyz(1), xyz(2));
170 return out;
171 }
